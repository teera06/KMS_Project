#include <vector>

using namespace std;

vector<int> solution(vector<int> arr, vector<int> query) {
    vector<int> answer;
    
    for(int i=0;i<query.size();i++){
        if(i%2==0){ // 짝수 인덱스 
            arr.erase(arr.begin()+(query[i]+1),arr.end());
        }else{ // 홀수 인덱스 
            arr.erase(arr.begin(),arr.begin()+(query[i]));
        }
    }
    
    return answer=arr;
}
